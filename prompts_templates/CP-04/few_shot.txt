Example 1:
Task:
Given an array of integers, return the sum of all positive differences between consecutive elements.

Solution:
<?php
function sumPositiveDiffs(array $nums): int {
    $sum = 0;
    for ($i = 1; $i < count($nums); $i++) {
        if ($nums[$i] > $nums[$i - 1]) {
            $sum += $nums[$i] - $nums[$i - 1];
        }
    }
    return $sum;
}
?>

Example 2:
Task:
Return the number of times a value increases compared to the previous value in an array.

Solution:
<?php
function countIncreases(array $nums): int {
    $count = 0;
    for ($i = 1; $i < count($nums); $i++) {
        if ($nums[$i] > $nums[$i - 1]) {
            $count++;
        }
    }
    return $count;
}
?>
